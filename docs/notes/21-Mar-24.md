```c
#include <stdio.h>

/*
    Bubble Sort
    Author atao
    3/21/2024
*/

void func(int arr[], int length) {
    for(int i = 0, tmp = 0; i < length-1; i++) {
        for(int n = 0; n < length-1-i; n++) {
            if (arr[n] > arr[n+1])
            {
                tmp = arr[n];
                arr[n] = arr[n+1];
                arr[n+1] = tmp;
            }
            
        }
    }
}

int main()
{
    int arr[6] = {5, 2, 4, 1, 6, 2};
    int length = sizeof(arr)/sizeof(arr[0]);
    printf("[");
    func(arr, length);
    for(int i = 0; i < length; i++) {
        if (i != length-1)
        {
            printf("\'%d\', ",arr[i]);
        } else {
            printf("\'%d\']\n", arr[i]);
        }
        
    }
    return 0;
}
```

```c
void func(int arr[], int length)
{
    for (int i = 0; i < length - 1; i++)
    {
        int *min_ptr = &arr[i];
        for (int n = i + 1; n < length; n++)
        {
            if (*min_ptr > arr[n])
            {
                min_ptr = &arr[n];
            }
        }
        int temp = arr[i];
        arr[i] = *min_ptr;
        *min_ptr = temp;
    }
}

int main()
{
    int arr[6] = {23, 1, 64, 788, 21, 67};
    int length = sizeof(arr) / sizeof(arr[0]);

    func(arr, length);

    printf("[");
    for (int i = 0; i < length; i++)
    {
        printf("%d", arr[i]);
        if (i != length - 1)
        {
            printf(", ");
        }
    }
    printf("]\n");

    return 0;
}
```

查找算法

二分法，定义一个左右指针，算出中指针，对比值

```c
#include <stdio.h>
#include <stddef.h>

/*
    Binary Search
    Author atao
    3/21/2024
*/

int func(int arr[], int length, int target)
{
    int *p_l = arr;
    int *p_r = arr + length - 1;
    while (p_l != p_r)
    {
        int distance = p_r - p_l;
        int *p_m = p_l + distance / 2;
        if (*p_m == target)
        {
            return p_m - arr;
        }
        else if (*p_m < target)
        {
            p_l = p_m + 1;
        }
        else
        {
            p_r = p_m - 1;
        }
    }
    return -1;
}

int main()
{
    int arr[10] = {13, 24, 35, 46, 57, 68, 79, 80, 91, 102};
    int length = sizeof(arr) / sizeof(arr[0]);
    int diff = func(arr, length, 35);
    if (diff == -1)
    {
        printf("Number not found.\n");
    }
    else
    {
        printf("Found number at index %d.\n", diff);
    }

    return 0;
}
```
